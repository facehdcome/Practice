/*
* 2.  
日本某地发生了一件谋杀案，警察通过排查确定杀人凶手必为4个
嫌疑犯的一个。以下为4个嫌疑犯的供词。
A说：不是我。
B说：是C。
C说：是D。
D说：C在胡说
已知3个人说了真话，1个人说的是假话。
现在请根据这些信息，写一个程序来确定到底谁是凶手。
假设凶手说假话，其他人说真话
*/
#include <stdio.h>
int main()
{
    int k = 0;
    for (k = 'a'; k <= 'd'; k++)
    {
        if (((k != 'a') + (k == 'c') + (k == 'd') + (k != 'd')) == 3)
            printf("k = %c", k);
    }
    return 0;
}



//3.在屏幕上打印杨辉三角。
/*
1
1 1
1 2 1
1 3 3 1
*/
int main()
{
	int arr[10][10] = { 0 };
	int i = 0;
	int j = 0;

	for (i = 0; i < 10; i++)
	{
		for (j = 0; j < 10; j++)
		{
			if (j == 0)
			{
				arr[i][j] = 1;
			}
			if (j == i)
			{
				arr[i][j] = 1;
			}
			if (i >= 2 && j >= 1)
			{
				arr[i][j] = arr[i - 1][j] + arr[i - 1][j - 1];
			}
		}
	}

	for (i = 0; i < 10; i++)
	{
		for (j = 0; j < 10 - 1 - i; j++)
		{
			printf("   ");
		}
		for (j = 0; j <= i; j++)
		{
			printf("%6d", arr[i][j]);
		}
		printf("\n");
	}

	return 0;
}
