//菱形打印二
#include <stdio.h>
int main()
{
	int line = 0;
	int i = 0;
	int j = 0;
	printf("请输入行数line:");
	scanf("%d", &line);

	for (i = 0; i < line; i++)						//上行
	{
		for (j = 0; j < line - 1 - i; j++)				//打印空格
		{
			printf(" ");
		}

		for (j = 0; j < 2 * i + 1; j++)				//打印星号
		{
			printf("*");
		}
		printf("\n");
	}

	for (i = 0; i < line - 1; i++)					//下行
	{
		//打印空格
		int j = 0;
		for (j = 0; j <= i; j++)
		{
			printf(" ");
		}

		for (j = 0; j < 2 * (line - 1 - i) - 1; j++)
		{
			printf("*");
		}
		printf("\n");
	}

	return 0;
}

{

		int n = 1;
		//每位数字至少有一位
		int temp = i;
		//将i的值赋给temp，防止修改i的循环变量
		int sum = 0; 
		//求和
		while (temp /= 10)
		{
			n++;
			//计算位数
		}
		temp = i;
		while (temp)
			//判断是否为0，是就退出
		{
			sum += (int)pow(temp % 10, n);
			temp = temp / 10;
		}
		if (i == sum)
		{
			printf(" %d ", i);
		}

	}

	return 0;
}




//写一个函数可以逆序字符串的内容
#include <assert.h>
void reverse(char* str)
{
	assert(str);
	int len = strlen(str);
	char* left = str;
	char* right = str + (len - 1);
	
	while (left < right)
	{
		char temp = *left;
		*left = *right;
		*right = temp;
		left++;
		right--;
	}
}

int main()
{
	char arr[100] = { 0 };
	printf("请输入字符串：\n");
	/*scanf("%s", arr);*/
	gets(arr);								//解决空格
	reverse(arr);
	printf("逆序排序后的结果是: %s", arr);

	return 0;
}
