//six_days
//2.写代码可以在整型有序数组中查找想要的数字，
//找到了返回下标，找不到返回 - 1.（折半查找）
#include <stdio.h>
int main()
{
    int arr[] = { 2,3,4,5,6,7,8,9,10,88,44,22,90 };
    int k = 7;
    int sz = sizeof(arr) / sizeof(arr[0]);

    int  left = 0;
    int right = sz-1;
    int mid = 0;

    while( left <= right)
    {
       mid = (left + right) / 2;
       if (arr[mid] == k)
       {
           printf("找到了！下标为: %d", mid);
           break;
       }
       else if (k < arr[mid])
           right = mid - 1;

       else if (k > arr[mid])
           left = mid + 1;
    }
    return 0;
}
